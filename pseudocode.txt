INITIALIZE N semaphores (forks)

// N Prozesse
FOR i IN 0 to N-1:
    PARALLEL:
        WHILE true:
            // Philosoph denkt
            DENKT RANDOME ZEIT

            // Philosoph wird hungrig
            IF i ist gerade:
                P(fork[i])         // left fork
                P(fork[(i+1) % N]) // right fork
            ELSE:
                P(fork[(i+1) % N]) // right fork
                P(fork[i])         // left fork

            // Philosoph isst
            ISST RANDOME ZEIT

            // Gabeln ablegen
            V(fork[i])         // left fork
            V(fork[(i+1) % N]) // right fork
